## 2. 封装一个客户端类

下面这个类封装了所有接口调用，**只用了 Python 标准库**，不需要额外装包。你可以直接复制到你的项目里用。

```python
"""OcrYoloEngine 接口客户端（纯标准库，零额外依赖）。"""

import base64
import json
import urllib.request
import urllib.error
from pathlib import Path


class OcrYoloClient:
    """调用 OcrYoloEngine 识别服务的客户端。"""

    def __init__(self, base_url: str = "http://localhost:8000", api_key: str = ""):
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key

    # ---------- 内部方法 ----------

    def _headers(self, content_type: str = "application/json") -> dict:
        h = {"Content-Type": content_type}
        if self.api_key:
            h["X-API-Key"] = self.api_key
        return h

    def _post_json(self, path: str, body: dict) -> dict:
        """发 POST 请求，返回解析后的 JSON 字典。"""
        data = json.dumps(body).encode()
        req = urllib.request.Request(
            f"{self.base_url}{path}",
            data=data,
            headers=self._headers(),
            method="POST",
        )
        try:
            with urllib.request.urlopen(req) as resp:
                return json.loads(resp.read())
        except urllib.error.HTTPError as e:
            error_body = json.loads(e.read()) if e.readable() else {}
            raise RuntimeError(
                f"HTTP {e.code}: {error_body.get('error_code', '')} - "
                f"{error_body.get('message', e.reason)}"
            ) from e

    def _get_json(self, path: str) -> dict:
        """发 GET 请求，返回解析后的 JSON 字典。"""
        req = urllib.request.Request(
            f"{self.base_url}{path}",
            headers=self._headers(),
        )
        with urllib.request.urlopen(req) as resp:
            return json.loads(resp.read())

    @staticmethod
    def _encode_image(image_path: str) -> str:
        """读取图片文件，返回 base64 编码字符串。"""
        return base64.b64encode(Path(image_path).read_bytes()).decode()

    # ---------- 识别接口 ----------

    def ocr(self, image_path: str) -> dict:
        """文字识别。

        参数:
            image_path: 图片文件路径

        返回:
            完整的响应字典
        """
        b64 = self._encode_image(image_path)
        return self._post_json("/v1/ocr", {"base64": b64})

    def detect(self, image_path: str, model: str, conf: float | None = None) -> dict:
        """目标检测（YOLO）。

        参数:
            image_path: 图片文件路径
            model:      模型名（在引擎的 configs/models.yaml 里登记的名字）
            conf:       置信度门槛（0~1），不传用引擎默认值 0.25

        返回:
            完整的响应字典
        """
        b64 = self._encode_image(image_path)
        body: dict = {
            "image": {"base64": b64},
            "methods": ["yolo"],
            "model": model,
        }
        if conf is not None:
            body["conf_threshold"] = conf
        return self._post_json("/v1/detect", body)

    def match(self, image_path: str, templates: list[str], conf: float | None = None) -> dict:
        """模板匹配。

        参数:
            image_path: 图片文件路径（大图，在里面找模板）
            templates:  要找的模板名列表（在引擎的 configs/templates.yaml 里登记的名字）
            conf:       置信度门槛（0~1）

        返回:
            完整的响应字典
        """
        b64 = self._encode_image(image_path)
        body: dict = {
            "image": {"base64": b64},
            "methods": ["template"],
            "templates": templates,
        }
        if conf is not None:
            body["conf_threshold"] = conf
        return self._post_json("/v1/match", body)

    def recognize(
        self,
        image_path: str,
        methods: list[str],
        model: str | None = None,
        templates: list[str] | None = None,
        conf: float | None = None,
        roi: dict | None = None,
        merge: str = "none",
        debug: bool = False,
    ) -> dict:
        """多方式混合识别（最灵活的接口）。

        参数:
            image_path: 图片文件路径
            methods:    识别方式列表，可选 "ocr" / "yolo" / "template"
            model:      用 yolo 时必填，模型名
            templates:  用 template 时必填，模板名列表
            conf:       置信度门槛
            roi:        只识别局部区域 {"x": 100, "y": 200, "w": 300, "h": 400}
            merge:      合并策略 "none" / "priority" / "dedup" / "concat"
            debug:      True 时额外返回标注图

        返回:
            完整的响应字典
        """
        b64 = self._encode_image(image_path)
        body: dict = {
            "image": {"base64": b64},
            "methods": methods,
            "merge": merge,
            "debug": debug,
        }
        if model:
            body["model"] = model
        if templates:
            body["templates"] = templates
        if conf is not None:
            body["conf_threshold"] = conf
        if roi:
            body["roi"] = roi
        return self._post_json("/v1/recognize", body)

    def upload(
        self,
        image_path: str,
        methods: list[str],
        model: str | None = None,
        templates: list[str] | None = None,
        conf: float | None = None,
    ) -> dict:
        """文件上传方式识别（不用 base64 编码，直接传文件）。

        参数同 recognize，但图片用文件上传而非 base64。
        """
        import mimetypes

        boundary = "----OcrYoloBoundary"
        body_parts = []

        # 图片文件
        mime = mimetypes.guess_type(image_path)[0] or "application/octet-stream"
        filename = Path(image_path).name
        body_parts.append(
            f'--{boundary}\r\n'
            f'Content-Disposition: form-data; name="file"; filename="{filename}"\r\n'
            f'Content-Type: {mime}\r\n\r\n'
        )
        file_data = Path(image_path).read_bytes()

        # 文本字段
        fields = {"methods": ",".join(methods)}
        if model:
            fields["model"] = model
        if templates:
            fields["templates"] = ",".join(templates)
        if conf is not None:
            fields["conf_threshold"] = str(conf)

        # 组装 multipart body
        parts = bytearray()
        parts.extend(body_parts[0].encode())
        parts.extend(file_data)
        parts.extend(b"\r\n")
        for key, val in fields.items():
            parts.extend(
                f'--{boundary}\r\n'
                f'Content-Disposition: form-data; name="{key}"\r\n\r\n'
                f'{val}\r\n'.encode()
            )
        parts.extend(f'--{boundary}--\r\n'.encode())

        req = urllib.request.Request(
            f"{self.base_url}/v1/recognize/upload",
            data=bytes(parts),
            headers={
                "Content-Type": f"multipart/form-data; boundary={boundary}",
                **({"X-API-Key": self.api_key} if self.api_key else {}),
            },
            method="POST",
        )
        with urllib.request.urlopen(req) as resp:
            return json.loads(resp.read())

    # ---------- 辅助方法：从响应里提取你要的数据 ----------

    @staticmethod
    def get_detections(resp: dict, method: str) -> list[dict]:
        """从响应里提取某个方式的检测结果列表。"""
        return resp.get("method_results", {}).get(method, {}).get("detections", [])

    @staticmethod
    def get_merged(resp: dict) -> list[dict]:
        """从响应里提取合并后的结果列表（用了 merge 参数时）。"""
        return resp.get("merged") or []

    # ---------- 管理接口 ----------

    def health(self) -> dict:
        """健康检查，返回 {"status": "ok"}。"""
        return self._get_json("/health")

    def ready(self) -> dict:
        """就绪检查，返回已加载的模型列表。"""
        return self._get_json("/ready")

    def list_models(self) -> list[dict]:
        """返回已登记的模型列表。"""
        return self._get_json("/v1/models").get("models", [])

    def list_templates(self) -> list[dict]:
        """返回已登记的模板列表。"""
        return self._get_json("/v1/templates").get("templates", [])

    def unload_model(self, name: str) -> dict:
        """卸载模型，释放内存/显存。"""
        return self._post_json(f"/v1/models/{name}/unload", {})

    def reload_model(self, name: str) -> dict:
        """重新加载模型（更新权重后调用）。"""
        return self._post_json(f"/v1/models/{name}/reload", {})
```

把上面的代码保存为 `ocr_yolo_client.py`，后面所有示例都基于它。

## 3. 模板匹配——找固定图标

适用场景：在截图里找某个**固定不变的图标**（比如设置按钮、关闭按钮、特定 Logo）。

### 接口信息

| 项目 | 值 |
|------|-----|
| 地址 | `POST /v1/match` |
| 传什么 | 截图（base64）+ 模板名列表 |
| 返回什么 | 每个找到的位置的坐标、置信度 |

### 调用代码

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

# ---- 找一个模板 ----
resp = client.match("screenshot.png", templates=["demo_block"])

# 拿结果
for det in client.get_detections(resp, "template"):
    x, y = det["center"]        # 中心点坐标 [x, y]
    score = det["confidence"]    # 把握程度 0~1
    name = det["label"]          # 匹配到的模板名
    print(f"找到 {name}，坐标 ({x}, {y})，把握 {score:.2f}")

# ---- 同时找多个模板 ----
resp = client.match("screenshot.png", templates=["login_btn", "close_icon"])
```

### 实际返回示例

```json
{
  "request_id": "a3f2...",
  "image_size": [160, 120],
  "method_results": {
    "template": {
      "detections": [
        {
          "source": "template",
          "label": "demo_block",
          "text": null,
          "confidence": 0.99,
          "bbox": [80.0, 60.0, 100.0, 80.0],
          "center": [90.0, 70.0],
          "bbox_norm": [0.5, 0.5, 0.625, 0.667],
          "center_norm": [0.5625, 0.5833]
        }
      ],
      "template_versions": {"demo_block": "v1"},
      "elapsed_ms": 2.1
    }
  }
}
```

### 关键字段说明

| 字段 | 说明 | 你的平台用哪个 |
|------|------|----------------|
| `center` | `[x, y]` 中心点像素坐标 | **点击用这个** |
| `bbox` | `[x1, y1, x2, y2]` 左上角+右下角像素坐标 | 画框用 |
| `confidence` | 0~1 | 判断是否可信 |
| `center_norm` | `[x, y]` 按比例（0~1） | 不同分辨率适配时用 |
| `label` | 模板名 | 区分找到的是哪个模板 |

## 4. 文字识别——找屏幕上的文字

适用场景：在截图里**找出所有文字**及其位置。

### 接口信息

| 项目 | 值 |
|------|-----|
| 地址 | `POST /v1/ocr` |
| 传什么 | 截图（base64） |
| 返回什么 | 每段文字的内容、坐标、置信度 |

> **注意**：`/v1/ocr` 的请求体格式和其他接口**不一样**——直接传 `{"base64": "..."}` 就行，不需要外面套 `image`。用上面的 `OcrYoloClient` 则不用管这个差异，已经封装好了。

### 调用代码

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

resp = client.ocr("screenshot.png")

for det in client.get_detections(resp, "ocr"):
    text = det["text"]            # 识别出的文字
    x, y = det["center"]          # 文字区域的中心点
    score = det["confidence"]
    print(f"文字「{text}」在 ({x}, {y})，把握 {score:.2f}")
```

### 常见用法：找特定文字的坐标

```python
def find_text(client, image_path, target_text):
    """在截图里找到包含指定文字的位置，返回中心坐标。"""
    resp = client.ocr(image_path)
    for det in client.get_detections(resp, "ocr"):
        if target_text in det["text"]:
            return det["center"]
    return None

pos = find_text(client, "screenshot.png", "登录")
if pos:
    print(f"「登录」按钮在 {pos}")
```

## 5. 目标检测——找特定物体

适用场景：在截图里找某类**物体**（比如游戏里的角色、道具）。需要引擎端登记过对应的 YOLO 模型。

### 接口信息

| 项目 | 值 |
|------|-----|
| 地址 | `POST /v1/detect` |
| 传什么 | 截图（base64）+ 模型名 |
| 返回什么 | 每个检测到的物体的类别名、坐标、置信度 |

### 调用代码

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

# 用引擎自带的通用模型 yolov8n（能识别人/车/动物等 80 类常见物体）
resp = client.detect("photo.jpg", model="yolov8n")

for det in client.get_detections(resp, "yolo"):
    label = det["label"]          # 类别名（如 "person", "car"）
    x, y = det["center"]
    score = det["confidence"]
    print(f"检测到 {label}，坐标 ({x}, {y})，把握 {score:.2f}")

# 只要把握程度高于 0.5 的结果
resp = client.detect("photo.jpg", model="yolov8n", conf=0.5)
```

## 6. 多方式混合识别

适用场景：一次请求**同时用多种方式识别**，比如既做 OCR 又做模板匹配。

### 接口信息

| 项目 | 值 |
|------|-----|
| 地址 | `POST /v1/recognize` |
| 传什么 | 截图 + 方式列表 + 各方式所需参数 |
| 返回什么 | 各方式的结果分组返回，可选合并 |

### 调用代码

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

# ---- 同时做 OCR + 模板匹配 ----
resp = client.recognize(
    "screenshot.png",
    methods=["ocr", "template"],
    templates=["login_btn", "close_icon"],
)

# 分别取各方式的结果
ocr_results = client.get_detections(resp, "ocr")
tpl_results = client.get_detections(resp, "template")

# ---- 用合并策略拿到一份统一列表 ----

# priority：按顺序试，有结果就停（模板匹配没找到就去 OCR 找）
resp = client.recognize(
    "screenshot.png",
    methods=["template", "ocr"],
    templates=["login_btn"],
    merge="priority",
)
merged = client.get_merged(resp)  # 只拿合并结果

# concat：所有方式的结果合在一起，按置信度排序
resp = client.recognize(
    "screenshot.png",
    methods=["ocr", "template"],
    templates=["login_btn"],
    merge="concat",
)
for det in client.get_merged(resp):
    print(f"[{det['source']}] {det.get('text') or det.get('label')} → {det['center']}")

# dedup：合在一起并去重（重叠超过 50% 的只保留置信度最高的）
resp = client.recognize(
    "screenshot.png",
    methods=["ocr", "template"],
    templates=["login_btn"],
    merge="dedup",
)
```

### 合并策略速查

| 策略 | 效果 | 适用场景 |
|------|------|----------|
| `none` | 各方式分开返回，`merged` 为空 | 需要分别处理各方式结果 |
| `priority` | 按顺序试，有结果就停 | "先用模板找，找不到再 OCR" |
| `concat` | 全部结果合在一起，按置信度排 | 只想拿一个列表 |
| `dedup` | 合在一起并去除重叠的重复项 | 多方式可能找到同一个目标 |

## 7. 文件上传方式

如果你不想做 base64 编码（比如文件比较大或者你用的语言处理 base64 麻烦），可以直接上传文件。

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

# 文件直接上传，不走 base64
resp = client.upload(
    "screenshot.png",
    methods=["template", "ocr"],
    templates=["login_btn"],
)
```

> 文件上传方式和 JSON 方式返回的结果格式**完全一样**。

## 8. 区域裁剪——只识别局部

如果你知道目标大概在截图的哪个区域（比如状态栏、底部导航），可以指定只识别那一块，**既省时间又排除干扰**。

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

# 只识别截图中 (100, 200) 位置开始、宽 300 高 400 的矩形区域
resp = client.recognize(
    "screenshot.png",
    methods=["ocr"],
    roi={"x": 100, "y": 200, "w": 300, "h": 400},
)

# 返回的坐标已经自动换算回原图坐标了，不用你自己加偏移
for det in client.get_detections(resp, "ocr"):
    print(f"「{det['text']}」在原图坐标 {det['center']}")
```

### ROI 参数说明

| 字段 | 类型 | 说明 |
|------|------|------|
| `x` | int | 区域左上角的 x 坐标（像素） |
| `y` | int | 区域左上角的 y 坐标（像素） |
| `w` | int | 区域宽度（像素） |
| `h` | int | 区域高度（像素） |

## 9. 调试标注图——看看识别到了什么

开发和调试时，可以让引擎返回一张**画了框的图片**，直观看到识别了什么。

```python
import base64
from pathlib import Path
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

resp = client.recognize(
    "screenshot.png",
    methods=["template"],
    templates=["demo_block"],
    debug=True,
)

# 把标注图保存到本地查看
if resp.get("debug_image"):
    img_bytes = base64.b64decode(resp["debug_image"])
    Path("debug_output.png").write_bytes(img_bytes)
    print("标注图已保存到 debug_output.png")
```

> `debug=True` 仅用于调试，生产环境关掉——它会额外消耗时间和带宽。

## 10. 资产管理接口

### 查看已登记的模板和模型

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

# 查看有哪些模板可用
templates = client.list_templates()
print("可用模板:", [t["name"] for t in templates])
# 输出: 可用模板: ['demo_block']

# 查看有哪些模型可用
models = client.list_models()
print("可用模型:", [m["name"] for m in models])
# 输出: 可用模型: ['yolov8n']
```

### 不重启更新模型

更新了模型文件后，不用重启服务，调一下 reload 就生效：

```python
# 卸载模型（释放内存/显存）
client.unload_model("yolov8n")

# 重新加载（加载新版本的权重文件）
client.reload_model("yolov8n")
```

## 11. 健康检查与监控

你的平台接入引擎后，建议定期做健康检查。

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

# 健康检查——引擎活着就返回 {"status": "ok"}
status = client.health()
is_alive = status.get("status") == "ok"

# 就绪检查——返回已加载的模型列表
ready = client.ready()
print(f"状态: {ready['status']}, 已配置模型: {ready['models']}")
```

### 监控指标（Prometheus 格式）

```python
import urllib.request

resp = urllib.request.urlopen("http://localhost:8000/metrics")
print(resp.read().decode())
```

返回的是纯文本，包含：
- `oye_requests_total{method="ocr"}` —— 各方式的请求次数
- `oye_inference_seconds_sum{method="template"}` —— 各方式的总推理耗时

## 12. 错误处理

引擎的所有错误都有统一格式。对接时建议统一处理。

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

try:
    resp = client.match("screenshot.png", templates=["some_template"])
except RuntimeError as e:
    error_msg = str(e)
    # 错误信息格式: "HTTP 404: TEMPLATE_NOT_FOUND - 模板 some_template 未注册"
    print(f"调用失败: {error_msg}")
```

### 错误码速查

| HTTP 状态码 | 错误码 | 什么意思 | 你的平台怎么处理 |
|---|---|---|---|
| 400 | `INVALID_IMAGE` | 图片无法解码 | 检查图片文件是否损坏 |
| 413 | `IMAGE_TOO_LARGE` | 图片超过大小限制 | 压缩后重试 |
| 403 | `PATH_NOT_ALLOWED` | 路径不在允许范围 | 改用 base64 方式传图 |
| 404 | `MODEL_NOT_FOUND` | 模型名没登记 | 检查 configs/models.yaml |
| 404 | `TEMPLATE_NOT_FOUND` | 模板名没登记 | 检查 configs/templates.yaml |
| 422 | — | 请求参数格式有问题 | 检查必填字段 |
| 503 | `OVERLOADED` | 引擎忙不过来 | 稍后重试 |
| 504 | `TIMEOUT` | 处理超时 | 图片太大或模型太慢 |
| 500 | `INTERNAL` | 引擎内部错误 | 报告给运维 |

### 建议的重试策略

```python
import time


def call_with_retry(fn, max_retries=3, backoff=1.0):
    """带重试的调用包装。遇到 503（太忙）自动等一下再试。"""
    for attempt in range(max_retries):
        try:
            return fn()
        except RuntimeError as e:
            if "HTTP 503" in str(e) and attempt < max_retries - 1:
                wait = backoff * (2 ** attempt)
                print(f"引擎繁忙，{wait}s 后重试...")
                time.sleep(wait)
            else:
                raise


# 用法
result = call_with_retry(lambda: client.match("screenshot.png", ["demo_block"]))
```

## 13. 返回数据结构速查

### 识别响应（所有识别接口通用）

```
{
  "request_id":     str      -- 请求唯一编号
  "image_size":     [w, h]   -- 原图宽高（像素）
  "method_results": {        -- 各方式的结果

      "detections": [...],   -- 检测结果列表

    "yolo": { ... }

  "merged":         [...] | null   -- 合并后的列表（用了 merge 时）
  "from_cache":     bool           -- 是否来自缓存
  "debug_image":    str | null     -- 标注图 base64（debug=true 时）
}
```

### 单个检测结果

```
{
  "source":      "ocr" | "yolo" | "template"  -- 来自哪种方式
  "label":       str | null    -- 类别名/模板名（OCR 时为空）
  "text":        str | null    -- 文字内容（非 OCR 时为空）
  "confidence":  float         -- 0~1 置信度
  "bbox":        [x1, y1, x2, y2]   -- 像素坐标框
  "center":      [cx, cy]           -- 像素坐标中心点 ← 你的平台点这里
  "bbox_norm":   [x1, y1, x2, y2]   -- 归一化坐标框（0~1）
  "center_norm": [cx, cy]           -- 归一化中心点（0~1）
}
```

# 取坐标（点击用）

## 完整对接示例

下面是一个完整的平台对接示例，演示真实的业务场景：

```python
"""平台对接 OcrYoloEngine 的完整示例。"""

from ocr_yolo_client import OcrYoloClient


def main():
    # 1. 初始化客户端
    client = OcrYoloClient(
        base_url="http://192.168.1.100:8000",   # 引擎服务地址
        api_key="your-secret-key",                # 有就填，没开验证就留空
    )

    # 2. 检查引擎是否在线
    if client.health().get("status") != "ok":
        print("引擎离线！")
        return

    # 3. 查看可用资产
    print("可用模板:", [t["name"] for t in client.list_templates()])
    print("可用模型:", [m["name"] for m in client.list_models()])

    # 4. 场景 A：在截图里找「登录」按钮的坐标
    resp = client.ocr("login_page.png")
    for det in client.get_detections(resp, "ocr"):
        if "登录" in (det["text"] or ""):
            print(f"登录按钮坐标: {det['center']}")
            break

    # 5. 场景 B：在截图里找关闭按钮图标
    resp = client.match("dialog.png", templates=["close_icon"])
    results = client.get_detections(resp, "template")
    if results:
        print(f"关闭按钮坐标: {results[0]['center']}")
    else:
        print("没找到关闭按钮")

    # 6. 场景 C：先找图标，找不到就找文字
    resp = client.recognize(
        "screenshot.png",
        methods=["template", "ocr"],
        templates=["submit_btn"],
        merge="priority",
    )
    merged = client.get_merged(resp)
    if merged:
        print(f"目标坐标: {merged[0]['center']}（来自 {merged[0]['source']}）")

    # 7. 场景 D：只看屏幕顶部状态栏的文字
    resp = client.recognize(
        "screenshot.png",
        methods=["ocr"],
        roi={"x": 0, "y": 0, "w": 1080, "h": 100},
    )
    for det in client.get_detections(resp, "ocr"):
        print(f"状态栏文字: {det['text']} @ {det['center']}")


if __name__ == "__main__":
    main()
```

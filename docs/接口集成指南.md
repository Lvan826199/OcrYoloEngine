## 1. Python 客户端类

下面这个类封装了所有接口，**只用 Python 标准库**，零额外依赖。复制到你的项目里就能用。

```python
"""OcrYoloEngine 接口客户端（纯标准库，零额外依赖）。"""

import base64
import json
import urllib.request
import urllib.error
from pathlib import Path


class OcrYoloClient:
    """调用 OcrYoloEngine 识别服务的客户端。"""

    def __init__(self, base_url: str = "http://localhost:8000", api_key: str = ""):
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key

    # ---------- 内部方法 ----------

    def _headers(self, content_type: str = "application/json") -> dict:
        h = {"Content-Type": content_type}
        if self.api_key:
            h["X-API-Key"] = self.api_key
        return h

    def _post_json(self, path: str, body: dict) -> dict:
        data = json.dumps(body).encode()
        req = urllib.request.Request(
            f"{self.base_url}{path}",
            data=data,
            headers=self._headers(),
            method="POST",
        )
        try:
            with urllib.request.urlopen(req) as resp:
                return json.loads(resp.read())
        except urllib.error.HTTPError as e:
            error_body = json.loads(e.read()) if e.readable() else {}
            raise RuntimeError(
                f"HTTP {e.code}: {error_body.get('error_code', '')} - "
                f"{error_body.get('message', e.reason)}"
            ) from e

    def _get_json(self, path: str) -> dict:
        req = urllib.request.Request(
            f"{self.base_url}{path}",
            headers=self._headers(),
        )
        with urllib.request.urlopen(req) as resp:
            return json.loads(resp.read())

    @staticmethod
    def _encode_image(image_path: str) -> str:
        return base64.b64encode(Path(image_path).read_bytes()).decode()

    # ---------- 识别接口 ----------

    def ocr(self, image_path: str) -> dict:
        """文字识别。"""
        b64 = self._encode_image(image_path)
        return self._post_json("/v1/ocr", {"base64": b64})

    def detect(self, image_path: str, model: str, conf: float | None = None) -> dict:
        """目标检测（YOLO）。model 为 configs/models.yaml 里登记的模型名。"""
        b64 = self._encode_image(image_path)
        body: dict = {"image": {"base64": b64}, "model": model}
        if conf is not None:
            body["conf_threshold"] = conf
        return self._post_json("/v1/detect", body)

    def match(self, image_path: str, templates: list[str], conf: float | None = None) -> dict:
        """模板匹配。templates 为 configs/templates.yaml 里登记的模板名列表。"""
        b64 = self._encode_image(image_path)
        body: dict = {"image": {"base64": b64}, "templates": templates}
        if conf is not None:
            body["conf_threshold"] = conf
        return self._post_json("/v1/match", body)

    def recognize(
        self,
        image_path: str,
        methods: list[str],
        model: str | None = None,
        templates: list[str] | None = None,
        conf: float | None = None,
        roi: dict | None = None,
        merge: str = "none",
        debug: bool = False,
    ) -> dict:
        """多方式混合识别。merge 可选 none/priority/dedup/concat。"""
        b64 = self._encode_image(image_path)
        body: dict = {"image": {"base64": b64}, "methods": methods, "merge": merge, "debug": debug}
        if model:
            body["model"] = model
        if templates:
            body["templates"] = templates
        if conf is not None:
            body["conf_threshold"] = conf
        if roi:
            body["roi"] = roi
        return self._post_json("/v1/recognize", body)

    def upload(
        self,
        image_path: str,
        methods: list[str],
        model: str | None = None,
        templates: list[str] | None = None,
        conf: float | None = None,
    ) -> dict:
        """文件上传方式识别（不用 base64，直接传文件）。"""
        import mimetypes

        boundary = "----OcrYoloBoundary"
        mime = mimetypes.guess_type(image_path)[0] or "application/octet-stream"
        filename = Path(image_path).name
        file_data = Path(image_path).read_bytes()
        fields = {"methods": ",".join(methods)}
        if model:
            fields["model"] = model
        if templates:
            fields["templates"] = ",".join(templates)
        if conf is not None:
            fields["conf_threshold"] = str(conf)
        parts = bytearray()
        parts.extend(
            f'--{boundary}\r\nContent-Disposition: form-data; name="file"; '
            f'filename="{filename}"\r\nContent-Type: {mime}\r\n\r\n'.encode()
        )
        parts.extend(file_data)
        parts.extend(b"\r\n")
        for key, val in fields.items():
            parts.extend(
                f'--{boundary}\r\nContent-Disposition: form-data; name="{key}"\r\n\r\n{val}\r\n'.encode()
            )
        parts.extend(f"--{boundary}--\r\n".encode())
        req = urllib.request.Request(
            f"{self.base_url}/v1/recognize/upload",
            data=bytes(parts),
            headers={
                "Content-Type": f"multipart/form-data; boundary={boundary}",
                **({"X-API-Key": self.api_key} if self.api_key else {}),
            },
            method="POST",
        )
        with urllib.request.urlopen(req) as resp:
            return json.loads(resp.read())

    # ---------- 辅助方法 ----------

    @staticmethod
    def get_detections(resp: dict, method: str) -> list[dict]:
        """从响应里提取某个方式的检测结果列表。"""
        return resp.get("method_results", {}).get(method, {}).get("detections", [])

    @staticmethod
    def get_merged(resp: dict) -> list[dict]:
        """从响应里提取合并后的结果列表（用了 merge 参数时）。"""
        return resp.get("merged") or []

    # ---------- 管理接口 ----------

    def health(self) -> dict:
        return self._get_json("/health")

    def ready(self) -> dict:
        return self._get_json("/ready")

    def list_models(self) -> list[dict]:
        return self._get_json("/v1/models").get("models", [])

    def list_templates(self) -> list[dict]:
        return self._get_json("/v1/templates").get("templates", [])

    def unload_model(self, name: str) -> dict:
        return self._post_json(f"/v1/models/{name}/unload", {})

    def reload_model(self, name: str) -> dict:
        return self._post_json(f"/v1/models/{name}/reload", {})
```

保存为 `ocr_yolo_client.py`，后面所有示例基于它。

## 2. 调用示例

> 每个接口的参数和返回字段的详细说明见 [使用文档](使用文档.md) 或浏览器打开 `http://localhost:8000/docs`。

### 模板匹配——找固定图标

```python
from ocr_yolo_client import OcrYoloClient

client = OcrYoloClient("http://localhost:8000")

resp = client.match("screenshot.png", templates=["demo_block"])

for det in client.get_detections(resp, "template"):
    x, y = det["center"]        # 中心点坐标——点击用这个
    score = det["confidence"]    # 把握程度 0~1
    name = det["label"]          # 匹配到的模板名
    print(f"找到 {name}，坐标 ({x}, {y})，把握 {score:.2f}")

# 同时找多个模板
resp = client.match("screenshot.png", templates=["login_btn", "close_icon"])
```

### 文字识别——找屏幕上的文字

```python
resp = client.ocr("screenshot.png")

for det in client.get_detections(resp, "ocr"):
    text = det["text"]           # 识别出的文字
    x, y = det["center"]        # 文字区域的中心点
    print(f"「{text}」在 ({x}, {y})")
```

**常见用法：找特定文字的坐标**

```python
def find_text(client, image_path, target_text):
    """在截图里找包含指定文字的位置，返回中心坐标。"""
    resp = client.ocr(image_path)
    for det in client.get_detections(resp, "ocr"):
        if target_text in det["text"]:
            return det["center"]
    return None

pos = find_text(client, "screenshot.png", "登录")
if pos:
    print(f"「登录」按钮在 {pos}")
```

### 目标检测——找特定物体

```python
# yolov8n 是仓库自带的通用模型，能识别人/车/动物等 80 类
resp = client.detect("photo.jpg", model="yolov8n")

for det in client.get_detections(resp, "yolo"):
    print(f"{det['label']}，坐标 {det['center']}，把握 {det['confidence']:.2f}")

# 只要置信度高于 0.5 的结果
resp = client.detect("photo.jpg", model="yolov8n", conf=0.5)
```

### 多方式混合识别 + 合并策略

```python
# 同时做 OCR + 模板匹配，各方式结果分开
resp = client.recognize(
    "screenshot.png",
    methods=["ocr", "template"],
    templates=["login_btn"],
)
ocr_results = client.get_detections(resp, "ocr")
tpl_results = client.get_detections(resp, "template")

# priority：按顺序试，有结果就停
resp = client.recognize(
    "screenshot.png",
    methods=["template", "ocr"],
    templates=["login_btn"],
    merge="priority",
)
merged = client.get_merged(resp)

# concat：全部结果合在一起按置信度排序
resp = client.recognize(
    "screenshot.png",
    methods=["ocr", "template"],
    templates=["login_btn"],
    merge="concat",
)
for det in client.get_merged(resp):
    print(f"[{det['source']}] {det.get('text') or det.get('label')} → {det['center']}")
```

> 合并策略的详细说明见 [使用文档 - 多方式合并](使用文档.md#7-多方式合并只在-v1recognize-生效)。

### 区域裁剪——只识别局部

## 3. 错误处理与重试

客户端类把 HTTP 错误统一转成 `RuntimeError`，消息格式：`HTTP 状态码: 错误码 - 描述`。

```python
try:
    resp = client.match("screenshot.png", templates=["some_template"])
except RuntimeError as e:
    # "HTTP 404: TEMPLATE_NOT_FOUND - 模板 some_template 未注册"
    print(f"调用失败: {e}")
```

> 全部错误码和含义见 [使用文档 - 错误码速查](使用文档.md#10-错误码速查)。
> 排查问题时记下响应里的 `request_id`（响应头 `X-Request-ID` 是同一个值），服务端日志可按它精确定位这次请求。

### 建议的重试策略（503 太忙时自动等一下再试）

```python
import time


def call_with_retry(fn, max_retries=3, backoff=1.0):
    for attempt in range(max_retries):
        try:
            return fn()
        except RuntimeError as e:
            if "HTTP 503" in str(e) and attempt < max_retries - 1:
                wait = backoff * (2 ** attempt)
                print(f"引擎繁忙，{wait}s 后重试...")
                time.sleep(wait)
            else:
                raise


result = call_with_retry(lambda: client.match("screenshot.png", ["demo_block"]))
```

## 4. 完整对接示例

一个完整的平台对接代码，演示 4 个真实业务场景：

```python
"""平台对接 OcrYoloEngine 的完整示例。"""

from ocr_yolo_client import OcrYoloClient


def main():
    client = OcrYoloClient(
        base_url="http://192.168.1.100:8000",
        api_key="your-secret-key",  # 没开验证就留空
    )

    # 检查引擎是否在线
    if client.health().get("status") != "ok":
        print("引擎离线！")
        return

    # 场景 A：找「登录」按钮的坐标
    resp = client.ocr("login_page.png")
    for det in client.get_detections(resp, "ocr"):
        if "登录" in (det["text"] or ""):
            print(f"登录按钮坐标: {det['center']}")
            break

    # 场景 B：找关闭按钮图标
    resp = client.match("dialog.png", templates=["close_icon"])
    results = client.get_detections(resp, "template")
    if results:
        print(f"关闭按钮坐标: {results[0]['center']}")

    # 场景 C：先找图标，找不到就找文字（priority 合并）
    resp = client.recognize(
        "screenshot.png",
        methods=["template", "ocr"],
        templates=["submit_btn"],
        merge="priority",
    )
    merged = client.get_merged(resp)
    if merged:
        print(f"目标坐标: {merged[0]['center']}（来自 {merged[0]['source']}）")

    # 场景 D：只看屏幕顶部状态栏
    resp = client.recognize(
        "screenshot.png",
        methods=["ocr"],
        roi={"x": 0, "y": 0, "w": 1080, "h": 100},
    )
    for det in client.get_detections(resp, "ocr"):
        print(f"状态栏: {det['text']} @ {det['center']}")


if __name__ == "__main__":
    main()
```
